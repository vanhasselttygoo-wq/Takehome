/* ---------------------------
   Matrix Rain Canvas Script
   ---------------------------
   - Adjust 'symbolSize', 'spawnRate', and 'speedFactor' below to tune look.
   - Canvas is overlayed and pointer-events are disabled so UI stays interactive.
*/
(function matrixRain(){
  const canvas = document.getElementById('matrixCanvas');
  const ctx = canvas.getContext('2d', {alpha: true});

  // Settings — tweak these to change density/speed/size
  let symbolSize = 18;        // font size in px
  let spawnRate = 0.95;       // probability each column spawns a new drop (0..1). Lower => fewer columns active
  let speedFactor = 0.6;      // global speed multiplier (0.2 slow, 1 normal, 2 fast)
  let fadeAlpha = 0.05;       // trail fade (0..1). Lower => longer trails

  const chars = 'アァカサタナハマヤャラワガザダバパイィキシチニヒミリヰギジヂビピウゥクスツヌフムユュルグズヅブプエェケセテネヘメレヱゲゼデベペオォコソトノホモヨョロゴゾドボポ0123456789@#$%&*+-=<>';

  let width, height, columns, drops;

  function resize(){
    width = canvas.width = window.innerWidth;
    height = canvas.height = window.innerHeight;
    columns = Math.floor(width / symbolSize);
    drops = new Array(columns).fill(0).map(() => ({
      y: Math.floor(Math.random()*-1000), // random start above screen
      speed: (Math.random()*0.9 + 0.6) * (symbolSize * speedFactor)
    }));
    ctx.font = `${symbolSize}px monospace`;
  }
  window.addEventListener('resize', resize, {passive:true});
  resize();

  // Draw loop
  function draw(){
    // fade background slightly to create trail
    ctx.fillStyle = `rgba(0, 0, 0, ${fadeAlpha})`;
    ctx.fillRect(0,0,width,height);

    // bright header glow occasionally
    for (let i=0;i<columns;i++){
      // randomly decide if this column is active (spawnRate)
      if (Math.random() > spawnRate && drops[i].y < 0) {
        // keep it quiet (helps density tuning)
        continue;
      }
      const x = i * symbolSize;
      const drop = drops[i];
      // set color gradient: bright head, faded tail
      const headChance = Math.random();
      // head glyph (brighter, larger alpha)
      ctx.fillStyle = `rgba(0,255,120,${0.95})`;
      const glyph = chars.charAt(Math.floor(Math.random() * chars.length));
      ctx.fillText(glyph, x, drop.y);

      // draw a secondary faint glyph slightly above for visual depth
      ctx.fillStyle = `rgba(0,200,100,0.35)`;
      const glyph2 = chars.charAt(Math.floor(Math.random() * chars.length));
      ctx.fillText(glyph2, x, drop.y - symbolSize);

      // step the drop
      drop.y += drop.speed;

      // wrap and randomize speed when it leaves bottom
      if (drop.y > height + Math.random() * 1000){
        drop.y = -Math.random() * 1000;
        drop.speed = (Math.random()*0.9 + 0.6) * (symbolSize * speedFactor);
      }
    }
    requestAnimationFrame(draw);
  }

  // Improve crispness on HiDPI screens
  function fixDPR(){
    const dpr = window.devicePixelRatio || 1;
    canvas.width = Math.floor(window.innerWidth * dpr);
    canvas.height = Math.floor(window.innerHeight * dpr);
    canvas.style.width = window.innerWidth + 'px';
    canvas.style.height = window.innerHeight + 'px';
    ctx.setTransform(dpr,0,0,dpr,0,0);
    ctx.font = `${symbolSize}px monospace`;
  }
  window.addEventListener('resize', () => { fixDPR(); resize(); }, {passive:true});
  fixDPR();

  // Start
  requestAnimationFrame(draw);

  // Optional: expose a global setter for quick runtime tweaks (for dev)
  window._matrixSettings = {
    setSize: s => { symbolSize = s; fixDPR(); resize(); },
    setDensity: r => { spawnRate = r; },
    setSpeed: v => { speedFactor = v; }
  };
})();

  if(!userName) return alert("Access denied: Enter alias first.");

};

function renderQuestion(){
  const q = questions[idx];
  qEl.textContent = (idx+1) + ". " + q.q;
  feedback.textContent = "";
  btnT.disabled = btnF.disabled = false;
  nextBtn.style.display = "none";
  updateProgress();
}

function updateProgress(){
  const percent = ((idx)/questions.length)*100;
  progressBar.style.width = percent + "%";
  progressText.textContent = `Question ${idx+1} of ${questions.length}`;
}

function answer(isTrue){
  const q = questions[idx];
  const correct = q.a === isTrue;
  userAnswers.push({question:q.q, answer:isTrue, correct});
  btnT.disabled = btnF.disabled = true;
  feedback.textContent = (correct?"✔ ACCESS GRANTED — ":"✖ ACCESS DENIED — ") + q.e;
  feedback.className = "feedback " + (correct?"correct":"wrong");
  nextBtn.style.display = "inline-block";
  if(correct) score++;
}

function next(){
  idx++;
  if(idx>=questions.length){
    progressBar.style.width = "100%";
    showSummary();
  }else renderQuestion();
}

function showSummary(){
  quizArea.style.display = "none";
  summary.style.display = "block";
  const percent = Math.round((score/questions.length)*100);
  scoreText.textContent = `${userName}, you scored ${score}/${questions.length} (${percent}%)`;
  let rating="";
  if(percent>=80){rating=">>> STATUS: ELITE HACKER 🟩";ratingEl.className="excellent";}
  else if(percent>=50){rating=">>> STATUS: INTERMEDIATE 🔶";ratingEl.className="good";}
  else{rating=">>> STATUS: RECRUIT 🔴";ratingEl.className="tryagain";}
  ratingEl.textContent = rating;
  progressText.textContent = "SYSTEM: Quiz Complete 🧠";

  fetch(SHEET_URL, {
    method: "POST",
    headers: {"Content-Type":"application/json"},
    body: JSON.stringify({
      name: userName,
      score: score,
      total: questions.length,
      percent: percent,
      answers: userAnswers
    })
  })
  .then(res => res.json())
  .then(d => saveStatus.textContent = "✔ Logs transmitted successfully")
  .catch(err => {
    console.error(err);
    saveStatus.textContent = "⚠ Transmission failed. Check Google Script config.";
  });
}

function restartQuiz(){
  idx=0;score=0;userAnswers=[];
  summary.style.display="none";
  quizArea.style.display="block";
  renderQuestion();
}

btnT.onclick=()=>answer(true);
